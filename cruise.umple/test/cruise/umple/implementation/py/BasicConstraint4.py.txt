class student():
    def __init__(self, aAge):
        self._age = aAge
        if 18 < = aAge :
            raise RuntimeError ("Please provide a valid age")

    def setAge(self, aAge):
        wasSet = False
        if 18 > aAge :
            self._age = aAge
            wasSet = True

        return wasSet

    def getAge(self):
        return self._age

    def delete(self):
        pass

    def __str__(self):
        return str(super()) + "[" + "age" + ":" + self.getAge() + "]"
