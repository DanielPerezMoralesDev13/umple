class Student:
    def __init__(self):
        self._grade = ""

    def setGrade(self, aGrade):
        wasSet = False
        self._grade = aGrade
        wasSet = True
        return wasSet

    def getGrade(self):
        return self._grade

    def delete(self):
        pass

    def __str__(self):
        return super().__str__() + "[" + "grade" + ":" + self.getGrade() + "]"
