class Client:
    def __init__(self, aMinAge, aAge):
        self._minAge = aMinAge
        self._age = aAge
        if aAge < = aMinAge :
            raise RuntimeError ("Please provide a valid age and minAge [age>minAge]")

    def setMinAge(self, aMinAge):
        wasSet = False
        if self.getAge() > aMinAge :
            self._minAge = aMinAge
            wasSet = True

        return wasSet

    def setAge(self, aAge):
        wasSet = False
        if aAge > self.getMinAge() :
            self._age = aAge
            wasSet = True

        return wasSet

    def getMinAge(self):
        return self._minAge

    def getAge(self):
        return self._age

    def delete(self):
        pass

    def __str__(self):
        return super().__str__() + "[" + "minAge" + ":" + self.getMinAge() + "," + "age" + ":" + self.getAge() + "]"
